闭包

```python
就是函数嵌套，内部函数可以引用外部函数的参数和变量，参数和变量不会被垃圾回收机制收回。
全局变量只有在页面关闭时才会回收
局部变量函数执行完就会回收
可以理解为闭包用用到的参数和变量都变为全局的。

function aaa(a){      
      var b = 5;      
      function bbb(){
           a++;
           b++;
         alert(a);
         alert(b);
      }
      return bbb;
  }

 var ccc = aaa(2); 
 ccc();
 ccc();
--------------------------------------------
封装成封闭函数
var ccc = (function(a){

  var b = 5;
  function bbb(){
       a++;
       b++;
     alert(a);
     alert(b);
  }
  return bbb;

})(2);

ccc();
ccc();
--------------------------------------------
作用1：将一个变量长期驻扎在内存当中
<script type="text/javascript">
    window.onload = function(){
```

面向对象

```python
javascript中的对象类似字典
--------------------------------------------
<script type="text/javascript">
var Tom = {
    name : 'tom',
    age : 18,
    showname : function(){
        alert('我的名字叫'+this.name);    
    },
    showage : function(){
        alert('我今年'+this.age+'岁');    
    }
}
</script>
--------------------------------------------
工厂模式
<script type="text/javascript">
    function Person(name){
        var o = new Object();
        o.name = name;
        o.showname = function(){
            alert('我的名字叫'+this.name);    
        };
        return o;
    }
    var tom = Person('tom');
    tom.showname();
</script>
--------------------------------------------
构造函数
<script type="text/javascript">
    function Person(name,age,job){            
        this.name = name;
        this.showname = function(){
            alert('我的名字叫'+this.name);    
        };
    }
    var tom = new Person('tom');
    var jack = new Person('jack');
    alert(tom.showjob==jack.showjob);
</script>
--------------------------------------------
原型模式 
<script type="text/javascript">
    function Person(name,age){        
        this.name = name;
        this.age = age;
    }
    Person.prototype.showname = function(){
        alert('我的名字叫'+this.name);    
    };
    Person.prototype.showage = function(){
        alert('我今年'+this.age+'岁');    
    };
    var tom = new Person('tom',18);
    var jack = new Person('jack',19);
    alert(tom.showname==jack.showname);
</script>
--------------------------------------------
继承
<script type="text/javascript">

    function fclass(name,age){
        this.name = name;
        this.age = age;
    }
    fclass.prototype.showname = function(){
        alert(this.name);
    }
    fclass.prototype.showage = function(){
        alert(this.age);
    }
    function sclass(name,age,job)
    {
        fclass.call(this,name,age);
        this.job = job;
    }
    sclass.prototype = new fclass();
    sclass.prototype.showjob = function(){
        alert(this.job);
    }
    var tom = new sclass('tom',19,'全栈工程师');
    tom.showname();
    tom.showage();
    tom.showjob();
</script>
```
