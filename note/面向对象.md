3.返回对象引用

class Person(object):
    def eat():
        print("eat")
	
    def run():
        print("run")
        
    # 私有的方法和属性不会被继承
    def __learn(self):
        print("learn")

    # 重写方法
    def run():
        print("run--------")

class Person(object):
    __instance = None

    def __new__(cls):
        if cls.__instance == None:
            cls.__instance = object.__new__(cls)
            return cls.__instance
        else:
            return cls.__instance
